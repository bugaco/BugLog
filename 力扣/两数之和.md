class Solution {
    func twoSum(_ nums: [Int], _ target: Int) -> [Int] {
        var map = [Int: Int]()
        for (index, num) in nums.enumerated() {
            if map.keys.contains(num) {
                return [map[num]!, index]
            } else {
                map[target - num] = index
            }
        }
        return []
    }
}

class Solution {
    func twoSum(_ nums: [Int], _ target: Int) -> [Int] {
        var map = [Int: Int]()
        for (i, item) in nums.enumerated() {
            for key in map.keys {
                if key + item == target {
                    return [map[key]!, i]

    }
